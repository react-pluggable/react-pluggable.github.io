Hello World

Let us create a plugin which will add alert functionality to the component.

Create a ClickMePlugin and implement it with IPlugin interface.

ClickMePlugin.tsx

import React from "react";
import { IPlugin, PluginStore } from "react-pluggable";

class ClickMePlugin implements IPlugin {
  public pluginStore: any;

  getPluginName(): string {
    return "ClickMe plugin";
  }
  getDependencies(): string[] {
    return [];
  }

  init(pluginStore: PluginStore): void {
    this.pluginStore = pluginStore;
  }

  activate(): void {
    this.pluginStore.addFunction("sendAlert", () => {
      alert("Testing");
    });
  }

  deactivate(): void {
    this.pluginStore.removeFunction("sendAlert");
  }
}

export default ClickMePlugin;

Now Add ClickMePlugin to your app

App.tsx

import React from "react";
import "./App.css";
import { createPluginStore, PluginProvider } from "react-pluggable";
import ClickMePlugin from "./components/plugins/ClickMePlugin";
import Test from "./components/Test";

function App() {
  const pluginStore = createPluginStore();
  pluginStore.install(new ClickMePlugin());

  pluginStore.addFunction("test", (a: any, b: any) => {
    console.log("working ", a, b);
  });

  pluginStore.addFunction("add", (a: number, b: number) => {
    console.log(a + b);
  });

  return (
    <PluginProvider pluginStore={pluginStore}>
      <Test />
    </PluginProvider>
  );
}

export default App;

Now using that plugin in our Target Component.

Test.tsx

import * as React from "react";
import { usePluginStore } from "react-pluggable";

const Test = () => {
  const pluginStore = usePluginStore();

  pluginStore.executeFunction("test", 1, 2);

  return (
    <>
      <h1>Working</h1>{" "}
      <button
        onClick={() => {
          pluginStore.executeFunction("sendAlert");
          pluginStore.executeFunction("add", 10, 20);
        }}
      >
        Send Alert
      </button>
    </>
  );
};

export default Test;
